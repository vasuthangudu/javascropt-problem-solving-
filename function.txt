   return a + b;
}

Default Parameters
function greet(name = "Guest"){
   console.log(name);
}

greet();

Rest Parameters
function total(...numbers){

   let sum = 0;

   for(let num of numbers){
      sum += num;
   }

   return sum;
}

console.log(total(1,2,3,4));
Spread Operator

function add(a,b,c){
   console.log(a+b+c);
}

Closures

Inner function remembers outer variables.

function outer(){

   let count = 0;

   function inner(){
      count++;
      console.log(count);
   }

   return inner;
}

const counter = outer();

counter();
counter();
counter();
Function Methods
1. call()
const person = {
   name: "Ram"
};

function greet(age){
   console.log(this.name, age);
}

greet.call(person, 22);
2. apply()

   name: "Sam"
};

function greet(city){
   console.log(this.name, city);
}

3. bind()
const person = {
   name: "Ravi"
};

function greet(){
   console.log(this.name);

const newFunc = greet.bind(person);

newFunc();
Real-Time Problems and Solutions
1. Check Even or Odd
function check(num){

   if(num % 2 === 0){
      return "Even";
   }

   return "Odd";
}

console.log(check(10));
2. Reverse a String
function reverseString(str){
   return str.split("").reverse().join("");
}

console.log(reverseString("hello"));
3. Largest Number in Array
function largest(arr){

   let max = arr[0];

   for(let num of arr){
      if(num > max){
         max = num;
      }
   }

   return max;
}

console.log(largest([10,40,20,80,5]));
4. Count Vowels
function countVowels(str){

   let count = 0;

   let vowels = "aeiouAEIOU";

   for(let char of str){
      if(vowels.includes(char)){
         count++;
      }
   }

   return count;
}

console.log(countVowels("JavaScript"));
5. Palindrome Check
function palindrome(str){

   let reversed = str.split("").reverse().join("");

   return str === reversed;
}

console.log(palindrome("madam"));
6. Sum of Array
function arraySum(arr){

   let sum = 0;

   for(let num of arr){
      sum += num;
   }

   return sum;
}

console.log(arraySum([1,2,3,4]));
7. Prime Number
function isPrime(num){

   if(num < 2){
      return false;
   }

   for(let i = 2; i < num; i++){

      if(num % i === 0){
         return false;
      }
   }

   return true;
}

console.log(isPrime(7));
8. Fibonacci Series
function fibonacci(n){

   let a = 0;
   let b = 1;

   console.log(a);
   console.log(b);

   for(let i = 2; i < n; i++){

      let c = a + b;

      console.log(c);

      a = b;
      b = c;
   }
}

fibonacci(6);
